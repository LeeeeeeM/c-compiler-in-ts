// while循环测试用例
int main() {
    int i;
    int sum;
    
    // 基本while循环
    i = 1;
    sum = 0;
    while (i <= 10) {
        sum = sum + i;
        i = i + 1;
    }
    printf("Sum from 1 to 10: %d\n", sum);
    
    // while循环中的if语句
    i = 1;
    while (i <= 5) {
        if (i % 2 == 0) {
            printf("Even: %d\n", i);
        } else {
            printf("Odd: %d\n", i);
        }
        i = i + 1;
    }
    
    // 嵌套while循环
    i = 1;
    while (i <= 3) {
        int j;  // 在while循环内部声明局部变量
        j = 1;
        while (j <= 3) {
            printf("i=%d, j=%d, product=%d\n", i, j, i * j);
            j = j + 1;
        }
        i = i + 1;
    }
    
    return 0;
}
